function checkCookie() { // Check for any of the possible valid cookies const validCookies = [ '24key', '48Key', '72key' ]; // Check if any valid cookie exists const hasValidCookie = validCookies.some(cookie => { return document.cookie.split('; ').some(c => c.startsWith(cookie + '=')); }); return hasValidCookie; } // Function to show error state function showErrorState() { document.getElementById('verificationUI').style.display = 'none'; document.getElementById('errorUI').style.display = 'block'; // Redirect after 2 seconds setTimeout(() => { window.location.href = "login.html"; }, 2000); } // Function to proceed with normal verification function proceedWithVerification() { const statusMessages = [ { text: "Valid cookie detected...", duration: 1500 }, { text: "Connecting to secure server...", duration: 2000 }, { text: "Loading PDF content...", duration: 2500 }, { text: "Finalizing verification...", duration: 1500 } ]; const progressBar = document.getElementById('progressBar'); const statusElement = document.getElementById('statusMessage'); let currentProgress = 0; let currentStatus = 0; function updateProgress() { if (currentProgress < 100) { currentProgress += Math.random() * 15 + 10; if (currentProgress > 100) currentProgress = 100; progressBar.style.width = `${currentProgress}%`; setTimeout(updateProgress, 500); } } function updateStatus() { if (currentStatus < statusMessages.length) { const { text, duration } = statusMessages[currentStatus]; statusElement.innerHTML = ` ${text}`; if (currentStatus === statusMessages.length - 1) { statusElement.style.fontWeight = '600'; statusElement.style.color = 'var(--primary-blue)'; } currentStatus++; setTimeout(updateStatus, duration); } else { statusElement.innerHTML = ` Verification complete! Loading PDF...`; progressBar.style.width = '100%'; // Redirect to PDF (replace with your actual URL) setTimeout(() => { window.location.href = "success.html"; }, 1000); } } updateProgress(); updateStatus(); } // Initial check when page loads if (checkCookie()) { // Cookie exists - proceed with normal verification setTimeout(proceedWithVerification, 1000); } else { // No valid cookie - show error state setTimeout(showErrorState, 1000); } // Prevent navigation (only if cookie exists) if (checkCookie()) { history.pushState(null, null, location.href); window.onpopstate = function() { history.go(1); const tempAlert = document.createElement('div'); tempAlert.style.position = 'fixed'; tempAlert.style.top = '20px'; tempAlert.style.left = '50%'; tempAlert.style.transform = 'translateX(-50%)'; tempAlert.style.backgroundColor = 'var(--error-red)'; tempAlert.style.color = 'white'; tempAlert.style.padding = '10px 20px'; tempAlert.style.borderRadius = '5px'; tempAlert.style.zIndex = '2000'; tempAlert.style.boxShadow = '0 3px 10px rgba(0,0,0,0.3)'; tempAlert.style.animation = 'fadeInOut 3s forwards'; tempAlert.innerHTML = ' Please do not navigate away during verification!'; document.body.appendChild(tempAlert); setTimeout(() => { tempAlert.remove(); }, 3000); }; window.onbeforeunload = function(e) { e.preventDefault(); e.returnValue = 'Your verification process will be interrupted. Are you sure you want to leave?'; return e.returnValue; }; } // Keyframe for fade animation const style = document.createElement('style'); style.textContent = ` @keyframes fadeInOut { 0%, 100% { opacity: 0; transform: translateX(-50%) translateY(-20px); } 10%, 90% { opacity: 1; transform: translateX(-50%) translateY(0); } } `; document.head.appendChild(style);
